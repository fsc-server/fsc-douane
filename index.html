const express = require('express'); const axios = require('axios'); const cors = require('cors'); const app = express(); app.use(cors()); app.use(express.json()); const CLIENT_ID = '1482588509054304396'; const CLIENT_SECRET = 'Ofqk6DnmYoeU4HotKzEtyfQiCA6Pplxs'; const GUILD_ID = '692145438198857859'; const REDIRECT_URI = process.env.REDIRECT_URI || 'https://fsc-douane-backend.onrender.com/callback'; const FRONTEND_URL = process.env.FRONTEND_URL || 'https://fsc-server.github.io/fsc-douane'; const ROLE_DOUANIER = '1201229701977485532'; const ROLES_RESPONSABLE = [ '1357186555680391241', '1249468622376534127', '1362786822786383923', '1243962143854956605' ]; // Étape 1 : rediriger vers Discord OAuth2 app.get('/login', (req, res) => { const params = new URLSearchParams({ client_id: CLIENT_ID, redirect_uri: REDIRECT_URI, response_type: 'code', scope: 'identify guilds.members.read' }); res.redirect(`https://discord.com/oauth2/authorize?${params}`); }); // Étape 2 : Discord renvoie un code, on l'échange contre un token app.get('/callback', async (req, res) => { const { code } = req.query; if (!code) return res.redirect(`${FRONTEND_URL}#error=no_code`); try { // Échanger le code contre un access token const tokenRes = await axios.post( 'https://discord.com/api/oauth2/token', new URLSearchParams({ client_id: CLIENT_ID, client_secret: CLIENT_SECRET, grant_type: 'authorization_code', code, redirect_uri: REDIRECT_URI }), { headers: { 'Content-Type': 'application/x-www-form-urlencoded' } } ); const accessToken = tokenRes.data.access_token; // Récupérer le profil Discord const userRes = await axios.get('https://discord.com/api/users/@me', { headers: { Authorization: `Bearer ${accessToken}` } }); const user = userRes.data; // Récupérer les rôles du membre sur le serveur FSC const memberRes = await axios.get( `https://discord.com/api/users/@me/guilds/${GUILD_ID}/member`, { headers: { Authorization: `Bearer ${accessToken}` } } ); const memberRoles = memberRes.data.roles || []; // Déterminer le rôle FSC let role = null; let roleid = null; for (const r of ROLES_RESPONSABLE) { if (memberRoles.includes(r)) { role = 'douanierplus'; roleid = r; break; } } if (!role && memberRoles.includes(ROLE_DOUANIER)) { role = 'douanier'; roleid = ROLE_DOUANIER; } if (!role) { return res.redirect(`${FRONTEND_URL}#error=no_role`); } const params = new URLSearchParams({ role, username: user.username, avatar: user.avatar || '', userid: user.id, roleid: roleid }); res.redirect(`${FRONTEND_URL}#${params}`); } catch (err) { console.error('OAuth error:', err.response?.data || err.message); res.redirect(`${FRONTEND_URL}#error=auth_failed`); } }); // Health check app.get('/', (req, res) => res.json({ status: 'FSC Backend OK' })); const PORT = process.env.PORT || 3000; app.listen(PORT, () => console.log(`FSC Backend running on port ${PORT}`));
